# List to store tasks (maximum size 6)
MAX_TASKS = 6
tasks = []

# Add Task
def add_task(name, description, start_time, end_time, repeat, priority):
    if len(tasks) >= MAX_TASKS:
        print("Task limit reached! Maximum 6 tasks allowed.")
        return

    if priority < 1 or priority > 5:
        print("Priority must be between 1 and 5.")
        return

    task = {
        "name": name,
        "description": description,
        "start_time": start_time,
        "end_time": end_time,
        "repeat": repeat,
        "priority": priority,
        "completed": False
    }

    tasks.append(task)
    print(f"Task added with ID: {len(tasks)}")


# Update Task
def update_task(task_id, name=None, description=None, start_time=None,
                end_time=None, repeat=None, priority=None, completed=None):

    if 1 <= task_id <= len(tasks):
        task = tasks[task_id - 1]

        if name is not None:
            task["name"] = name
        if description is not None:
            task["description"] = description
        if start_time is not None:
            task["start_time"] = start_time
        if end_time is not None:
            task["end_time"] = end_time
        if repeat is not None:
            task["repeat"] = repeat
        if priority is not None:
            if 1 <= priority <= 5:
                task["priority"] = priority
            else:
                print("Priority must be between 1 and 5.")
                return
        if completed is not None:
            task["completed"] = completed

        print("Task updated successfully.")
    else:
        print("Task ID not found.")


# Remove Task
def remove_task(task_id):
    if 1 <= task_id <= len(tasks):
        tasks.pop(task_id - 1)
        print("Task removed successfully.")
    else:
        print("Task ID not found.")


# Retrieve All Tasks
def get_all_tasks():
    if not tasks:
        print("No tasks available.")
        return

    for index, task in enumerate(tasks):
        print(f"\nID: {index + 1}")
        print(f"Name: {task['name']}")
        print(f"Description: {task['description']}")
        print(f"Start Time: {task['start_time']}")
        print(f"End Time: {task['end_time']}")
        print(f"Repeat: {task['repeat']}")
        print(f"Priority: {task['priority']}")
        print(f"Completed: {task['completed']}")


# Retrieve Single Task
def get_task(task_id):
    if 1 <= task_id <= len(tasks):
        task = tasks[task_id - 1]
        print(f"\nID: {task_id}")
        print(f"Name: {task['name']}")
        print(f"Description: {task['description']}")
        print(f"Start Time: {task['start_time']}")
        print(f"End Time: {task['end_time']}")
        print(f"Repeat: {task['repeat']}")
        print(f"Priority: {task['priority']}")
        print(f"Completed: {task['completed']}")
    else:
        print("Task ID not found.")
